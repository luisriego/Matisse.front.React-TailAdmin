interface DecodedToken {
  iat: number;
  exp: number;
  roles: string[];
  username: string;
  id: string;
  name: string;
  unit: string | null;
}

        setUserEmail(decodedToken.username); // Corrected from decodedToken.user

        console.error('Erro ao decodificar token:', error);
